Numerical experiments on cosh-weighted Hilbert transform¶

This notebook is to study the computational characteristics of the Bertola-Katsevich-Tovbis inversion formulas for consh-weighted Hilbert transform that solves the image reconstruciton problem in several practical imaging applications.

There are two pairs of known funcitons and their cosh-weighted Hilbert transforms. These two pairs are used to study the numerical behavior of the inversion formulas on error propagation and noise response. In particular, it is necessary to find out the range of $\mu$ for the numerical inversion remains accurate and stable to noise.

def test_sin(t, mu):
    f = np.sin(mu*np.sqrt(1-t*t))
    F = np.sinh(mu*t)
    return f, F

def test_cos(t, mu):
    f = (np.cos(mu*np.sqrt(1-t*t))*np.sqrt(1-t*t))
    F = (t*np.cosh(mu*t)-0.5*mu*np.sinh(mu*t))
    return f, F

def hilbert_cos(dim):
    _grid = np.arange(0.5, dim, 1)*(np.pi/dim)
    _cos = np.cos(_grid)
    
    ht_matrix = np.empty((dim, dim))
    ii = 0
    while ii < dim:
        jj = 0
        while jj < dim:
            if jj == ii:
                ht_matrix[ii, jj] = 0.0
            elif jj > ii:
                ht_matrix[ii, jj] = step/(_cos[jj] - _cos[ii])
            else:
                ht_matrix[ii, jj] = -ht_matrix[jj, ii]
            jj += 1
        ii += 1

    return ht_matrix

step = np.pi/dim
t_step = np.arange(0+step/2.0, np.pi, step)
c_step = np.cos(t_step)
s_step = np.sin(t_step)

ht_matrix = hilbert_cos(dim)
cos_w = np.cos(mu*s_step)
sin_w = np.sin(mu*s_step)

if test_func == 1:
    f_0, fht = test_cos(c_step, mu)
else:
    f_0, fht = test_sin(c_step, mu)

F_0 = np.copy(fht)

np.random.seed(11)
noise = np.random.normal(0.0, sigma, dim)
F_1 = fht + (noise + 1j*noise)
F_mse = 2*np.mean(np.square(noise))
print("source noise: sigma = {0}, mse = {1:.3f}".format(sigma, F_mse))

source noise: sigma = 0.2, mse = 0.081

fht_cos = np.multiply(cos_w, F_1)

fht_cos = np.multiply(cos_w, fht_cos)
fht_cos = np.multiply(s_step, fht_cos)

fht_sin = np.multiply(sin_w, F_1)
fht_sin = np.multiply(s_step, fht_sin)
fht_sin = np.matmul(ht_matrix, fht_sin)
fht_sin = np.multiply(sin_w, fht_sin)

f_1 = (fht_cos + fht_sin)/np.pi

diff = f_1-f_0
diff = np.multiply(diff, np.conj(diff))
f_mse = np.mean(np.abs(diff))
print("inverse noise: mse = {:.3f}".format(f_mse))
